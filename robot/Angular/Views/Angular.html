public class aaaRepository : IaaaRepository
{
    TBSAdminEntities dbContext = new TBSAdminEntities();
    IaaaDal _aaaDal;
    protected const int BatchSize = 10000;
    protected string TbsConnectionString = ConfigurationManager.ConnectionStrings["TBSConnection"].ConnectionString;
    protected string TbsAdminConnectionString = ConfigurationManager.ConnectionStrings["TBSAdminConnection"].ConnectionString;
    protected string DestinationTableName = "PromoEligibleClient";
    ///////////////////////////////////////////////////////////////////////////////////////////
    // automapper setup
    public aaaRepository(IaaaDal _aaaDal)
    {
        AutoMapper.Mapper.CreateMap();
        AutoMapper.Mapper.CreateMap();
        AutoMapper.Mapper.CreateMap();
        this._aaaDal = _aaaDal;
    }
    ///////////////////////////////////////////////////////////////////////////////////////////
    // CRUD functions for aaa 
    public IEnumerableGetaaa()
    {
        return dbContext.aaaList().Map();
    }
    public aaaUpdateResult Updateaaa(aaa aaa)
    {
        return dbContext.aaaUpdate(
            aaa.id,
            aaa.name,
            aaa.email,
            aaa.dob,
            aaa.groupID,
            aaa.cool,
        ).Map().FirstOrDefault();
    }
    public int Insertaaa(aaa aaa)
    {
        var Result = dbContext.aaaInsert(
            aaa.id,
            aaa.name,
            aaa.email,
            aaa.dob,
            aaa.groupID,
            aaa.cool,
        );
        return Result.Select(x => x.aaaID).FirstOrDefault().Value;
    }
    public aaaDeleteResult Deleteaaa(int id)
    {
        return dbContext.aaaDelete(id).Map().FirstOrDefault();
    }
}
